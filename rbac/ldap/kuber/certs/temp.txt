openssl genrsa -out rootCAkey.pem 2048


openssl req -x509  -new -nodes   -key rootCAkey.pem   -days 3650   -out cacerts.pem -subj "/CN=rahul"


cfssl gencert -ca=cacerts.pem \
-ca-key=rootCAkey.pem \
-config=ca-config.json \
-profile=server kafka-server-domain.json | cfssljson -bare kafka-server



openssl x509 -in kafka-server.pem -text -noout

sudo kubectl create secret generic tls-kafka --from-file=fullchain.pem=kafka-server.pem --from-file=cacerts.pem=cacerts.pem --from-file=privkey.pem=kafka-server-key.pem -n confluent


kubectl create secret generic credential \
  --from-file=plain-users.json=creds-kafka-sasl-users.json \
  --from-file=plain.txt=creds-client-kafka-sasl-user.txt \
  --from-file=ldap.txt=ldap.txt \
  --namespace confluent


kubectl create secret generic mds-token \
  --from-file=mdsPublicKey.pem=publickey.txt \
  --from-file=mdsTokenKeyPair.pem=tokenkeypair.txt \
  --namespace confluent

  kubectl create secret generic mds-client \
  --from-file=bearer.txt=bearer.txt \
  --namespace confluent
